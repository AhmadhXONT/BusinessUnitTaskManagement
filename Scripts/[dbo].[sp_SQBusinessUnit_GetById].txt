/*

ALTER PROCEDURE [dbo].[sp_SQBusinessUnit_GetById]
    @Id INT
AS
BEGIN
    SET NOCOUNT ON;
    SELECT 
    Id,
    BusinessUnit AS BusinessUnitCode,
    BusinessUnitName,
    AddressLine1,
    TelephoneNumber,
    IsActive,
    CreatedBy,
    CreatedOn,
    Logo,
